#include <tunables/global>

profile huly flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability,
  file,
  signal,
  mount,
  umount,
  remount,
  pivot_root,

  # S6-Overlay
  /init ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /etc/fix-attrs.d/** rw,
  /etc/s6/** ixr,
  /etc/s6-rc/** ixr,
  /run/{s6,s6-rc*,service}/** rwix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/lib/bashio/** ix,
  /etc/s6/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,

  # Docker CLI + Compose — run in THIS profile, not a nested child. HAOS 18.1's
  # kernel (6.18) denies AF_UNIX socket connects from processes confined by
  # nested child profiles regardless of the rules they contain (verified with
  # parser 3.1.7 and 4.1.7: the identical ruleset works flat, fails nested).
  # Flat matches the portainer/arcane/dockhand profiles, which were unaffected.
  /usr/bin/docker ix,
  /usr/libexec/docker/** ix,

  # Standalone Docker Compose binary (bypasses plugin loader segfault on aarch64)
  /usr/local/bin/docker-compose ix,
  /usr/lib/docker/cli-plugins/** ix,
  /usr/libexec/docker/cli-plugins/** ix,

  # Access to Docker socket — the Supervisor mounts it at /run/docker.sock and
  # /var/run is a symlink to /run; AppArmor matches the resolved path, so the
  # rule must cover /run/docker.sock (HAOS 18.1+ enforces this)
  /run/docker.sock rw,
  /var/run/docker.sock rw,

  # Network access (all families — AF_UNIX is required for the Docker socket)
  network,
}
